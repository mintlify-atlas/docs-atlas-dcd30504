## Overview

`@beztack/mercadopago` provides a comprehensive SDK for integrating Mercado Pago payments and subscriptions into your Beztack application. It includes:

- **Server-side client** for API interactions
- **React components** for payments and subscriptions UI
- **React Query hooks** for data fetching and mutations
- **Type-safe schemas** with Zod validation
- **i18n support** for Spanish and English

## Installation

<CodeGroup>
```bash npm
npm install @beztack/mercadopago
```

```bash pnpm
pnpm add @beztack/mercadopago
```

```bash yarn

```
</CodeGroup>

## Server-Side SDK

Import from `@beztack/mercadopago/server` to use the server-side client.

### Creating a Client

```typescript
import { createMercadoPagoClient } from "@beztack/mercadopago/server"

const mp = createMercadoPagoClient({
  accessToken: process.env.MERCADO_PAGO_ACCESS_TOKEN!,
  webhookSecret: process.env.MERCADO_PAGO_WEBHOOK_SECRET,
  timeout: 30000,        // 30 seconds (default)
  maxRetries: 3,         // Retry 429/5xx errors (default: 3)
  initialRetryDelay: 1000, // 1 second (default)
  integratorId: "your_integrator_id", // Optional
})
```

<ParamField path="accessToken" type="string" required>
  Your Mercado Pago access token
</ParamField>

<ParamField path="webhookSecret" type="string">
  Secret for webhook signature validation
</ParamField>

<ParamField path="baseUrl" type="string" default="https://api.mercadopago.com">
  API base URL (for testing)
</ParamField>

<ParamField path="timeout" type="number" default={30000}>
  Request timeout in milliseconds
</ParamField>

<ParamField path="maxRetries" type="number" default={3}>
  Maximum retry attempts for 429/5xx errors
</ParamField>

<ParamField path="initialRetryDelay" type="number" default={1000}>
  Initial retry delay in milliseconds (exponential backoff)
</ParamField>

<ParamField path="integratorId" type="string">
  Mercado Pago partner metrics tracking ID
</ParamField>

### Payments Module

#### Get Payment

```typescript
const payment = await mp.payments.get("123456789")

console.log(payment.status)           // "approved"
console.log(payment.transaction_amount) // 100
console.log(payment.payer?.email)      // "user@example.com"
```

<ResponseField name="id" type="number">
  Payment ID
</ResponseField>

<ResponseField name="status" type="string">
  Payment status: `pending`, `approved`, `authorized`, `in_process`, `in_mediation`, `rejected`, `cancelled`, `refunded`, `charged_back`
</ResponseField>

<ResponseField name="transaction_amount" type="number">
  Payment amount
</ResponseField>

<ResponseField name="currency_id" type="string">
  Currency code (e.g., "UYU", "USD", "ARS")
</ResponseField>

<ResponseField name="payer" type="object">
  <ResponseField name="email" type="string">Payer email</ResponseField>
  <ResponseField name="id" type="string">Payer ID</ResponseField>
</ResponseField>

### Webhooks Module

#### Validate Webhook

```typescript
// In your webhook handler
const isValid = mp.webhooks.validate(
  xSignature,  // x-signature header
  xRequestId,  // x-request-id header
  dataId       // data.id from webhook payload
)

if (!isValid) {
  return new Response("Invalid signature", { status: 401 })
}
```

<Tip>
In development (when `NODE_ENV !== "production"`), webhook validation is skipped if `webhookSecret` is not configured.
</Tip>

#### Parse Webhook

```typescript
const payload = mp.webhooks.parse(rawBody)

console.log(payload.type)   // "payment" | "subscription" | etc.
console.log(payload.action) // "created" | "updated" | etc.
console.log(payload.data.id) // Resource ID
```

### Error Handling

The SDK throws `MercadoPagoError` instances:

```typescript
import { MercadoPagoError } from "@beztack/mercadopago/server"

try {
  const payment = await mp.payments.get("123")
} catch (error) {
  if (error instanceof MercadoPagoError) {
    console.error(error.statusCode)  // HTTP status code
    console.error(error.message)     // Error message
    console.error(error.errorCause)  // Original error
    console.error(error.retryable)   // Whether error is retryable
  }
}
```

## React SDK

Import from `@beztack/mercadopago/react` to use React components and hooks.

### Setup

Wrap your app with `MercadoPagoProvider`:

```tsx
import { MercadoPagoProvider } from "@beztack/mercadopago/react"

function App() {
  return (
    <MercadoPagoProvider
      publicKey={import.meta.env.VITE_MERCADO_PAGO_PUBLIC_KEY}
      apiBaseUrl={import.meta.env.VITE_API_URL}
      locale="es-UY"
    >
      <YourApp />
    </MercadoPagoProvider>
  )
}
```

<ParamField path="publicKey" type="string" required>
  Mercado Pago public key
</ParamField>

<ParamField path="apiBaseUrl" type="string" required>
  Your API base URL (for calling your backend endpoints)
</ParamField>

<ParamField path="locale" type="MercadoPagoLocale" default="es-UY">
  Locale for Mercado Pago SDK: `es-UY`, `es-AR`, `es-CL`, `es-CO`, `es-MX`, `es-VE`, `es-PE`, `pt-BR`, `en-US`
</ParamField>

### Plans Hooks

#### usePlans

Fetch subscription plans with optional filtering:

```tsx
import { usePlans } from "@beztack/mercadopago/react"

function PlansList() {
  const { data, isLoading, error } = usePlans({
    status: "active",
    enabled: true,
  })

  if (isLoading) return <Spinner />
  if (error) return <Error message={error.message} />

  return (
    <ul>
      {data?.plans.map(plan => (
        <li key={plan.id}>
          {plan.reason} - {plan.transactionAmount} {plan.currencyId}
        </li>
      ))}
    </ul>
  )
}
```

<ResponseField name="data" type="PlansResponse">
  <ResponseField name="plans" type="Plan[]">Array of plans</ResponseField>
  <ResponseField name="total" type="number">Total count</ResponseField>
  <ResponseField name="syncedAt" type="string | null">Last sync timestamp</ResponseField>
</ResponseField>

#### usePlan

Fetch a single plan by ID:

```tsx
import { usePlan } from "@beztack/mercadopago/react"

function PlanDetail({ planId }: { planId: string }) {

```

### Payment Events Hook

#### usePaymentEvents

Real-time payment events via Server-Sent Events (SSE):

```tsx
import { usePaymentEvents } from "@beztack/mercadopago/react"

function PaymentEventsMonitor() {
  const { events, connectionStatus, clearEvents, reconnect } = usePaymentEvents({
    enabled: true,
    onEvent: (event) => {
      console.log("New event:", event.type, event.resourceId)
      toast.info(`${event.type}: ${event.action}`)
    },
    onError: (error) => {
      console.error("SSE error:", error)
    },
    reconnectInterval: 5000,
    maxRetries: 5,
  })

  return (
    <div>
      <p>Status: {connectionStatus}</p>
      <button onClick={reconnect}>Reconnect</button>
      <button onClick={clearEvents}>Clear</button>
      
      <ul>
        {events.map(event => (
          <li key={event.id}>
            {event.type} - {event.action} - {event.resourceId}
          </li>
        ))}
      </ul>
    </div>
  )
}
```

<ParamField path="enabled" type="boolean" default={true}>
  Enable/disable the SSE connection
</ParamField>

<ParamField path="onEvent" type="(event: PaymentEvent) => void">
  Callback for each event received
</ParamField>

<ParamField path="onError" type="(error: Error) => void">
  Callback for errors
</ParamField>

<ParamField path="reconnectInterval" type="number" default={5000}>
  Reconnection interval in milliseconds
</ParamField>

<ParamField path="maxRetries" type="number" default={5}>
  Maximum reconnection attempts
</ParamField>

<ResponseField name="events" type="PaymentEvent[]">
  Array of received events (max 100)
</ResponseField>

<ResponseField name="connectionStatus" type="ConnectionStatus">
  Current connection status: `connecting`, `connected`, `disconnected`
</ResponseField>

<ResponseField name="clearEvents" type="() => void">
  Clear all stored events
</ResponseField>

<ResponseField name="reconnect" type="() => void">
  Manually reconnect
</ResponseField>

### Server-Side Webhook Handler

```typescript
import { createMercadoPagoClient } from "@beztack/mercadopago/server"
import { defineEventHandler, readRawBody } from "h3"

const mp = createMercadoPagoClient({
  accessToken: process.env.MERCADO_PAGO_ACCESS_TOKEN!,
  webhookSecret: process.env.MERCADO_PAGO_WEBHOOK_SECRET,
})

export default defineEventHandler(async (event) => {
  const xSignature = event.headers.get("x-signature")
  const xRequestId = event.headers.get("x-request-id")
  const rawBody = await readRawBody(event)
  
  if (!rawBody) {
    throw createError({ statusCode: 400, message: "No body" })
  }

  const payload = mp.webhooks.parse(rawBody)
  const dataId = payload.data.id

  // Validate signature
  const isValid = mp.webhooks.validate(xSignature, xRequestId, dataId)
  if (!isValid) {
    throw createError({ statusCode: 401, message: "Invalid signature" })
  }

  // Handle different event types
  if (payload.type === "payment") {
    const payment = await mp.payments.get(dataId)
    
    if (payment.status === "approved") {
      // Process approved payment
      await processPayment(payment)
    }
  } else if (payload.type === "subscription") {
    const subscription = await mp.subscriptions.get(dataId)
    
    // Update subscription in database
    await updateSubscription(subscription)
  }

  return { received: true }
})
```

### Query Key Management

```tsx
import { plansKeys, subscriptionsKeys } from "@beztack/mercadopago/react"
import { useQueryClient } from "@tanstack/react-query"

function AdminDashboard() {
  const queryClient = useQueryClient()

  const handlePlanCreated = () => {
    // Invalidate all plans queries
    queryClient.invalidateQueries({ queryKey: plansKeys.all })
  }

  const handleSubscriptionUpdated = (subscriptionId: string) => {
    // Invalidate specific subscription
    queryClient.invalidateQueries({
      queryKey: subscriptionsKeys.detail(subscriptionId)
    })
    
    // Invalidate all subscription lists
    queryClient.invalidateQueries({
      queryKey: subscriptionsKeys.lists()
    })
  }

  return <Dashboard />
}
```
