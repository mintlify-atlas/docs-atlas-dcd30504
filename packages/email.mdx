# Email Package

The `@beztack/email` package provides type-safe email sending with React Email templates and Resend delivery. It includes pre-built templates for common transactional emails.

## Installation

The email package is optional. Add it during project creation:

```bash Terminal
pnpm dlx beztack init --modules email
```

Or install manually:

```bash Terminal
pnpm add @beztack/email --filter @beztack/api
```

## Available Templates

### Welcome Email

Sent to new users after registration.

```typescript Welcome
import { sendEmail } from "@beztack/email"

await sendEmail({
  type: "welcome",
  to: "user@example.com",
  data: {
    username: "John Doe",
    loginUrl: "https://app.example.com/login",
  },
})
```

**Required Data:**

<ParamField path="username" type="string" required>
  User's display name
</ParamField>

<ParamField path="loginUrl" type="string" required>
  URL to login page
</ParamField>

### Password Reset

Sent when users request a password reset.

```typescript Reset
import { sendEmail } from "@beztack/email"

await sendEmail({
  type: "password-reset",
  to: "user@example.com",
  data: {
    username: "John Doe",
    resetUrl: "https://app.example.com/reset?token=abc123",
  },
})
```

**Required Data:**

<ParamField path="username" type="string" required>
  User's display name
</ParamField>

<ParamField path="resetUrl" type="string" required>
  Password reset link with token
</ParamField>

### Subscription Confirmation

Sent after successful subscription purchase.

```typescript Subscription
import { sendEmail } from "@beztack/email"

await sendEmail({

  to: "user@example.com",

    username: "John Doe",
    planName: "Pro Plan",
    amount: "$29/month",
    billingPeriod: "monthly",
    dashboardUrl: "https://app.example.com/dashboard",
  },
})
```

**Required Data:**

<ParamField path="username" type="string" required>
  User's display name
</ParamField>

<ParamField path="planName" type="string" required>
  Subscription plan name
</ParamField>

<ParamField path="amount" type="string" required>
  Formatted price (e.g., "$29/month")
</ParamField>

<ParamField path="billingPeriod" type="string" required>
  Billing period ("monthly", "yearly")
</ParamField>

<ParamField path="dashboardUrl" type="string" required>
  URL to user dashboard
</ParamField>

**Required Data:**

## API Reference

### `sendEmail()`

```typescript Function Signature
type EmailType =
  | "welcome"
  | "password-reset"
  | "subscription-confirmation"
  | "organization-invitation"

interface SendEmailUnifiedProps {
  type: EmailType
  to: string | string[]
  data: Record<string, unknown>
}

interface EmailResult {
  success: boolean
  data?: {
    id: string
    message: string
  }
  error?: string
}

function sendEmail(props: SendEmailUnifiedProps): Promise<EmailResult>
```

**Parameters:**

<ParamField path="type" type="EmailType" required>
  Template type to use
</ParamField>

<ParamField path="to" type="string | string[]" required>
  Recipient email address(es)
</ParamField>

<ParamField path="data" type="object" required>
  Template-specific data (see template sections above)
</ParamField>

**Returns:**

<ResponseField name="success" type="boolean">
  Whether email was sent successfully
</ResponseField>

<ResponseField name="data" type="object">
  Email delivery information (only if success is true)
  
  - `id`: Resend email ID
  - `message`: Success message
</ResponseField>

<ResponseField name="error" type="string">
  Error message (only if success is false)
</ResponseField>

## Usage in API Routes

### Nitro Route Handler

```typescript apps/api/server/routes/api/auth/send-reset.ts
import { sendEmail } from "@beztack/email"
import { defineEventHandler, readBody, createError } from "h3"

export default defineEventHandler(async (event) => {
  const { email, resetToken } = await readBody(event)

  const result = await sendEmail({
    type: "password-reset",
    to: email,
    data: {
      username: "User",
      resetUrl: `https://app.example.com/reset?token=${resetToken}`,
    },
  })

  if (!result.success) {
    throw createError({
      statusCode: 500,
      message: result.error || "Failed to send email",
    })
  }

  return { message: "Reset email sent" }
})
```

### With Database Query

```typescript Complete Example
import { sendEmail } from "@beztack/email"
import { db, user } from "@beztack/db"
import { eq } from "drizzle-orm"
import { defineEventHandler, createError } from "h3"

export default defineEventHandler(async (event) => {
  const userId = event.context.user?.id

  if (!userId) {
    throw createError({ statusCode: 401, message: "Unauthorized" })
  }

  const [currentUser] = await db
    .select()
    .from(user)
    .where(eq(user.id, userId))

  if (!currentUser) {
    throw createError({ statusCode: 404, message: "User not found" })
  }

  const result = await sendEmail({
    type: "welcome",
    to: currentUser.email,
    data: {
      username: currentUser.name,
      loginUrl: "https://app.example.com/login",
    },
  })

  return { success: result.success }
})
```

## Configuration

Set up Resend API credentials:

```bash .env
RESEND_API_KEY=re_xxxxxxxxxxxx
RESEND_FROM_EMAIL=noreply@yourdomain.com
RESEND_FROM_NAME=Your App Name
```

<Note>
  Get your Resend API key from [resend.com/api-keys](https://resend.com/api-keys) and verify your domain.
</Note>

## Creating Custom Templates

Add new email templates:

**1. Create React component:**

```tsx packages/email/emails/custom.tsx
import {
  Body,
  Container,
  Head,
  Heading,
  Html,
  Link,
  Preview,
  Text,
} from "@react-email/components"

interface CustomEmailProps {
  userName: string
  actionUrl: string
}

export function CustomEmail({ userName, actionUrl }: CustomEmailProps) {
  return (
    <Html>
      <Head />
      <Preview>Custom email preview text</Preview>
      <Body style={{ backgroundColor: "#ffffff", fontFamily: "sans-serif" }}>
        <Container>
          <Heading>Hello {userName}</Heading>
          <Text>This is a custom email template.</Text>
          <Link href={actionUrl}>Take Action</Link>
        </Container>
      </Body>
    </Html>
  )
}
```

**2. Update `sendEmail` types** in `packages/email/lib/send.ts`
