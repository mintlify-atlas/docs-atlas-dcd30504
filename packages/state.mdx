# State Package

The `@beztack/state` package provides type-safe URL state management using nuqs. It allows you to sync component state with URL search parameters, enabling shareable links and browser navigation support.

## Installation

The state package is optional. Add it during project creation:

```bash Terminal
pnpm dlx beztack init --modules state
```

Or install manually:

```bash Terminal
pnpm add @beztack/state --filter @beztack/ui
```

## Usage Examples

### Basic String State

```typescript Component
import { useQueryState, parseAsString } from "@beztack/state"

function SearchBox() {
  const [search, setSearch] = useQueryState(
    "q",
    parseAsString.withDefault("")
  )

  return (
    <input
      type="text"
      value={search}
      onChange={(e) => setSearch(e.target.value)}
      placeholder="Search..."
    />
  )
}

// URL: ?q=hello
```

### Number State

```typescript Pagination
import { useQueryState, parseAsInteger } from "@beztack/state"

function Pagination() {
  const [page, setPage] = useQueryState(
    "page",
    parseAsInteger.withDefault(1)
  )

  return (
    <div>
      <button type="button" onClick={() => setPage(page - 1)}>Previous</button>
      <span>Page {page}</span>
      <button type="button" onClick={() => setPage(page + 1)}>Next</button>
    </div>
  )
}

// URL: ?page=3
```

### Boolean State

```typescript Toggle
import { useQueryState, parseAsBoolean } from "@beztack/state"

function FilterPanel() {
  const [showFilters, setShowFilters] = useQueryState(
    "filters",
    parseAsBoolean.withDefault(false)
  )

  return (
    <div>
      <button
        type="button"
        onClick={() => setShowFilters(!showFilters)}
      >
        Toggle Filters
      </button>
      {showFilters && <div>Filter options...</div>}
    </div>
  )
}

// URL: ?filters=true
```

### Enum State

```typescript Select
import { useQueryState, parseAsStringEnum } from "@beztack/state"

function SortSelector() {
  const [sort, setSort] = useQueryState(
    "sort",
    parseAsStringEnum(["name", "date", "price"] as const).withDefault("name")
  )

  return (
    <select value={sort} onChange={(e) => setSort(e.target.value)}>
      <option value="name">Name</option>
      <option value="date">Date</option>
      <option value="price">Price</option>
    </select>
  )
}

// URL: ?sort=date
```

### Array State

```typescript Multi-select
import { useQueryState, parseAsArrayOf, parseAsString } from "@beztack/state"

function TagFilter() {
  const [tags, setTags] = useQueryState(
    "tags",
    parseAsArrayOf(parseAsString).withDefault([])
  )

  const toggleTag = (tag: string) => {
    setTags(
      tags.includes(tag)
        ? tags.filter((t) => t !== tag)
        : [...tags, tag]
    )
  }

  return (
    <div>
      <button type="button" onClick={() => toggleTag("react")}>React</button>
      <button type="button" onClick={() => toggleTag("typescript")}>TypeScript</button>
      <button type="button" onClick={() => toggleTag("vite")}>Vite</button>
      <p>Selected: {tags.join(", ")}</p>
    </div>
  )
}

// URL: ?tags=react&tags=typescript
```

### Multiple States

```typescript useQueryStates
import { useQueryStates, parseAsString, parseAsInteger } from "@beztack/state"

function ProductList() {
  const [filters, setFilters] = useQueryStates({
    search: parseAsString.withDefault(""),
    category: parseAsString.withDefault("all"),
    page: parseAsInteger.withDefault(1),
    limit: parseAsInteger.withDefault(10),
  })

  return (
    <div>
      <input
        type="text"
        value={filters.search}
        onChange={(e) => setFilters({ search: e.target.value })}
      />
      <select
        value={filters.category}
        onChange={(e) => setFilters({ category: e.target.value })}
      >
        <option value="all">All</option>
        <option value="electronics">Electronics</option>
      </select>
      <p>
        Page {filters.page} of {Math.ceil(100 / filters.limit)}
      </p>
    </div>
  )
}

// URL: ?search=laptop&category=electronics&page=2&limit=20
```

### JSON State with Validation

```typescript Complex State
import { useQueryState, parseAsJson } from "@beztack/state"
import { z } from "zod"

const filterSchema = z.object({
  minPrice: z.number(),
  maxPrice: z.number(),
  inStock: z.boolean(),
})

type Filters = z.infer<typeof filterSchema>

function AdvancedFilters() {
  const [filters, setFilters] = useQueryState(
    "filters",
    parseAsJson(filterSchema.parse).withDefault({
      minPrice: 0,
      maxPrice: 1000,
      inStock: false,
    })
  )

  return (
    <div>
      <input
        type="number"
        value={filters.minPrice}
        onChange={(e) =>
          setFilters({ ...filters, minPrice: Number(e.target.value) })
        }
      />
      <input
        type="checkbox"
        checked={filters.inStock}
        onChange={(e) =>
          setFilters({ ...filters, inStock: e.target.checked })
        }
      />
    </div>
  )
}

// URL: ?filters={"minPrice":50,"maxPrice":500,"inStock":true}
```

## Next.js Integration

### App Router Setup

Wrap your app with the adapter:

```tsx app/layout.tsx
import { NuqsAdapter } from "@beztack/state/adapters/next/app"

export default function RootLayout({
  children,
}: {
  children: React.ReactNode
}) {
  return (
    <html lang="en">
      <body>
        <NuqsAdapter>{children}</NuqsAdapter>
      </body>
    </html>
  )
}
```

### Server-Side Usage

```typescript Server Component
import { createSearchParamsCache } from "@beztack/state/server"
import { parseAsString, parseAsInteger } from "@beztack/state"
import type { SearchParams } from "@beztack/state/server"

const searchParamsCache = createSearchParamsCache({
  q: parseAsString.withDefault(""),
  page: parseAsInteger.withDefault(1),
})

type PageProps = {
  searchParams: Promise<SearchParams>
}

export default async function Page({ searchParams }: PageProps) {
  const { q, page } = await searchParamsCache.parse(searchParams)

  return (
    <div>
      <h1>Search: {q}</h1>
      <p>Page {page}</p>
    </div>
  )
}
```

## API Reference

### `useQueryState()`

```typescript Signature
function useQueryState<T>(
  key: string,
  parser?: Parser<T>,
  options?: UseQueryStateOptions
): [T, (value: T | null) => void]
```

<ParamField path="key" type="string" required>
  URL parameter name
</ParamField>

<ParamField path="parser" type="Parser<T>">
  Value parser with default
</ParamField>

<ParamField path="options" type="object">
  - `history`: `"push"` | `"replace"` (default: `"push"`)
  - `shallow`: Skip re-render on change
  - `scroll`: Scroll to top on change
  - `throttleMs`: Throttle state updates
</ParamField>

### `useQueryStates()`

```typescript Signature
function useQueryStates<T extends Record<string, Parser<any>>>(
  parsers: T,
  options?: UseQueryStatesOptions
): [Values<T>, (updates: Partial<Values<T>>) => void]
```

Manage multiple query parameters at once.
